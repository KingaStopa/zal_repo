class NWD
	def liczby(a,b)
		@a = a
		@b = b
		liczby_reguly
	end

	def liczby_reguly
		if @a > @b
			@wieksza_liczba = @a
			@mniejsza_liczba = @b
		else
			@wieksza_liczba = @b
			@mniejsza_liczba = @a
		end
	end

	def obliczenia
		reszta = 1
		dzielna = @wieksza_liczba
		dzielnik = @mniejsza_liczba
		until reszta == 0
			reszta = dzielna % dzielnik
			dzielna = dzielnik
			dzielnik = reszta
		end
		dzielna
	end
end
